crontab -e

*/5 * * * * /Applications/XAMPP/bin/php /Applications/XAMPP/xamppfiles/htdocs/Auction_WebApp/cron/check_expired_auctions.php >> /Applications/XAMPP/xamppfiles/htdocs/Auction_WebApp/cron/cron.log 2>&1

cat /Applications/XAMPP/xamppfiles/htdocs/Auction_WebApp/cron/cron.log

php -v

cd /Applications/XAMPP/xamppfiles/htdocs/Auction_WebApp
curl -sS https://getcomposer.org/installer | /Applications/XAMPP/bin/php

ls -la
